void switchsetup(int sw,int pos){
  digitalWrite(switches[sw], pos);
  positions[sw] = pos;
}
double getVoltage(){
    double maxValue = 0;
    double minValue = 4095;
    for (int i = 0; i < samples; i++) {
      int val = analogRead(SENSOR_PIN);
      if (val > maxValue) maxValue = val;
      if (val < minValue) minValue = val;
    }
    double peakToPeak = maxValue - minValue;
    double voltage = (peakToPeak / 2.0) * (3.3 / 4095.0);
    double acRMS = voltage * 0.707 * calibration_factor;
    delay(1000);
    return acRMS;
}

double getCurrent1(){
  if(positions[1] == 1){
    int mx1 = 0;int mn1 = 4096;
    uint32_t st_tm1 = millis();
    while((millis()-st_tm1) < 500){ //sample for 1 Sec
      int rv1 = analogRead(c1);
      if (rv1 > mx1){mx1 = rv1;}
      if (rv1 < mn1){mn1 = rv1;}
    }
    double vo1 = ((mx1 - mn1) * 3.3)/4096.0; //ESP32 ADC resolution 4096
    double vr1 = (vo1/2.0) *0.707;   //root 2 is 0.707
    return ((vr1 * 1000)/mVperAmp);
  }else{
    return 0.00;
  }
}

void implimentAi(){
  //Condition 1
  //Appliance Auto-OFF on No Occupancy
  if(humanState==0){
    if(lightstatestart==0){
      lightstatestart=millis();
    }else{
      if(millis()>lightstatestart+(5*60*1000)){
        switchsetup(1,0);
        switchsetup(3,0);
        switchsetup(2,0);
        switchsetup(4,0);
        lightstatestart=0;
      }
    }
  }else{
       lightstatestart=0;
  }
  //Condition 2
  //Day/Night Lighting Control
  if(lightState==1){
      if(positions[1]==1){
        switchsetup(1,0);
      }
      if(positions[2]==1){
        switchsetup(2,0);
      } 
  }else{
    if(humanState == 1 && positions[1]==0){
      switchsetup(1,1);
    }
  }
  //Condition 3
  //Temperature-Based Fan Control
  if(temprature>15){
    if(humanState==1){
      switchsetup(3,1);
    }else{
      switchsetup(3,0);
    }
  }else{
       switchsetup(3,0);
  }
  //Condition 4
  //Gas Leakage Protection
  if(gasState==0){
    switchsetup(4,1);
    switchsetup(1,0);
    switchsetup(2,0);
    switchsetup(3,0);
    //Blynk.notify("LPG Leakage,Please Check");
    Blynk.logEvent("gas_leakage");
  }else{
    switchsetup(4,0);
  }
//condition 5-6
//Sudden Overcurrent Protection / High Power Consumtion Protection
  if(currents[1]>safelimit){
    switchsetup(1,0);
  }
  if(currents[2]>safelimit){
    switchsetup(2,0);
  }
  if(currents[3]>safelimit){
    switchsetup(3,0);
  }
  if(currents[4]>safelimit){
    switchsetup(4,0);
  }
}
